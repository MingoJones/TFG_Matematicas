\begin{definition}
    Se define un polinomio trigonométrico como una función $P:\mathbb{T} \to \mathbb{C}$ de la forma
    \[
        P(x) = \sum_{k=-M}^M c_k e^{2\pi i k x}
    \]
    con $c_k \in \mathbb{C}$ y $M \in \mathbb{N}$.
\end{definition}

\begin{theorem}[Teorema de Aproximación de Weierstrass]
    \label{ap:weierstrass}
    Sea $f: \mathbb{T} \to \mathbb{C}$ continua y sea $\varepsilon>0$, entonces
    existe un polinomio trigonométrico $P$ tal que
    \[
        \sup_{x\in \mathbb{T}} \left|f(x)-P(x)\right| \leq \varepsilon
    \]
\end{theorem}

\begin{definition}
    \label{ap:bessel}
    La función de Bessel de orden $\bl{n} \in \mathbb{Z}$, denotada por $J_n(x)$, se
    define como el $n$-ésimo coeficiente de Fourier de la funcion $e^{ix\sin(\theta)}$, es decir,
    \[
        J_n(x) = \frac{1}{2\pi} \int_{0}^{2\pi} e^{ix\sin(\theta)} e^{-in\theta} d\theta, \quad x\in\mathbb{R}.
    \]
\end{definition}
Esta función la utilizamos en el ejemplo \ref{ex:sin_noeq}, con los valores $n=0$ y $x=2\pi$.

\begin{theorem}[Desigualdad de Cauchy-Schwarz]
    \label{ap:cauchy-schwarz}
    Sean $a_1, a_2, \ldots, a_n$ y $b_1, b_2, \ldots, b_n \in \mathbb{C}$ entonces
    \[
        \left(\sum_{k=1}^n a_k\overline{b_k}\right) \leq \left(\sum_{k=1}^n |a_k|^2\right) \left(\sum_{k=1}^n |b_k|^2\right)
    \]

\end{theorem}
