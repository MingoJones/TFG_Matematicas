\begin{proposition}
    Sea $(x_n)_{n\geq1}$ una sucesión de números reales.
    Las siguientes afirmaciones son equivalentes:
    \begin{enumerate}
        \item $(x_n)$ es equidistribuida módulo $1$.

        \item La igualdad (\ref{def:equidistribucion}) se cumple
        para todo intervalo $(a,b)\subseteq \mathbb{T}$.

        \item La igualdad (\ref{def:equidistribucion}) se cumple
        para todo intervalo $[a,b], \,[a,b), \,(a,b] \subseteq \mathbb{T}$.

        \item La igualdad (\ref{def:equidistribucion}) se cumple
        para todo intervalo $[0,b)\subseteq \mathbb{T}$.

        \item La igualdad (\ref{def:equidistribucion}) se cumple
        para todo intervalo $(a,1)\subseteq \mathbb{T}$.

        \item La igualdad (\ref{def:equidistribucion}) se cumple
        para todo intervalo $(a,b)$ con $a,b\in D$, donde $D \subseteq \mathbb{T}$ es un conjunto denso.
    \end{enumerate}
\end{proposition}

\end{proof}
