\begin{theorem}[Teorema de Fejér]
    \label{teo:fejer_euler}
    Sea $g:[1,\infty) \to \mathbb{R}$ una función derivable tal que $g' \in C([1,\infty))$ y $g'(x)$ 
    es monótona para $x \geq x_0$ para algún $x_0 \geq 1$. Si se cumplen las siguientes condiciones,
    \begin{align}
        \label{hipo:fejer_euler}
        \lim_{x \to \infty} g'(x) = 0
        \qquad \text{y} \qquad
        \lim_{x \to \infty} {x \left|g'(x)\right|} = \infty 
    \end{align}
    entonces la sucesión $(g(n))_{n\geq1}$ es equidistribuida módulo $1$.
\end{theorem}

\begin{proof}
    Sea $N > x_0$ y sea $k \in \mathbb{Z}\setminus\{0\}$. 
    Aplicando la fórmula (\ref{prop:formula_euler}) a $f(x) = e^{2\pi i k g(x)}$,
    \[
        \sum_{n=1}^N e^{2\pi i k g(n)}
        = \int_1^N e^{2\pi i k g(x)} \, dx 
        + \frac{1}{2} \left(e^{2\pi i k g(N)} + e^{2\pi i k g(1)}\right)
        + \int_1^N \left(\partfrac{x} - \frac{1}{2}\right) 2\pi i k g'(x) \, e^{2\pi i k g(x)} \, dx
    \]
    La primera integral la separamos en los dos intervalos $[1,x_0]$ y $[x_0,N]$,
    \[
        \int_1^N e^{2\pi i k g(x)} \,dx = 
        \int_1^{x_0} e^{2\pi i k g(x)} \, dx + 
        \int_{x_0}^N e^{2\pi i k g(x)} \, dx
    \]
    % Aqui puedo salvar esta parte de la monotonia y solo necesitar que sea monotona en x>=x_0
    El primer intervalo se acota por
    \[
        \left|\int_1^{x_0} e^{2\pi i k g(x)} \, dx\right| \leq x_0 - 1
    \]
    y al dividir por $N$, se anula cuando $N \to \infty$.
    
    Para el segundo, aplicamos integración por partes,
    \begin{align*}
        \int_{x_0}^N e^{2\pi i k g(x)} \, dx 
        = \int_{x_0}^N \frac{\left(e^{2\pi i k g(x)}\right)'}{2\pi i k g'(x)} \, dx
        &= \frac{1}{2 \pi i k} \int_{x_0}^N \frac{\left(e^{2\pi i k g(x)}\right)'}{g'(x)} \, dx \\
        &= \frac{1}{2 \pi i k} \left(\left[\frac{e^{2\pi i k g(x)}}{g'(x)}\right]_{x_0}^N - 
        \int_{x_0}^N e^{2\pi i k g(x)} \left(\frac{1}{g'(x)}\right)' \, dx \right)
    \end{align*}
    % El primer termino no tiene problema con la monotonia.
    Vamos a acotar cada uno de los sumandos. En el primero,
    \[
        \left|\frac{e^{2\pi i k g(N)}}{g'(N)} - \frac{e^{2\pi i k g(x_0)}}{g'(x_0)}\right|
        \leq \frac{1}{|g'(N)|} + \frac{1}{|g'(x_0)|} 
    \]
    Dividiendo por $N$ y usando la hipótesis (\ref{hipo:fejer_euler}), se tiene que
    \[
        \frac{1}{N |g'(N)|} + \frac{1}{N |g'(x_0)|} \longrightarrow 0 \qquad \text{cuando} \ N \to \infty
    \]
    
    Para el segundo, tenemos
    \[
        \left|\int_{x_0}^N e^{2\pi i k g(x)}\left(\frac{1}{g'(x)}\right)' \, dx\right|
        \leq \int_{x_0}^N \left|\left(\frac{1}{g'(x)}\right)'\right| \, dx
    \]
    % Aqui hay problemas con g' 
    Como $g'(x)$ es monótona a partir de $x_0$ y $\lim_{x \to \infty} g'(x) = 0$, 
    entonces $g'(x)$ no cambia de signo para $x \geq x_0$. Además, no puede ser que 
    $g'(x) = 0$ para algún $x \geq x_0$, ya que entonces, por ser monótona, $g'(x) = 0$ 
    a partir de ahí, lo que contradice la condición $\lim_{x \to \infty} x|g'(x)| = \infty$.
    
    Por lo tanto, $1/g'(x)$ es monótona para $x \geq x_0$, luego $(1/g'(x))'$ es de 
    signo constante para $x \geq x_0$. 
    \[
        \int_{x_0}^N \left|\left(\frac{1}{g'(x)}\right)'\right| \, dx =
        \left|\int_{x_0}^N \left(\frac{1}{g'(x)}\right)' \, dx\right| =
        \left|\frac{1}{g'(N)} - \frac{1}{g'(x_0)} \right| \leq 
        \frac{1}{|g'(N)|} + \frac{1}{|g'(x_0)|}
    \]
    y al igual que antes, al dividir por $N$ se anula cuando $N \to \infty$.

    Todo esto muestra que,
    \[
        \lim_{N \to \infty} \frac{1}{N} \int_1^N e^{2\pi i k g(x)} \, dx = 0
    \]

    El término central se acota por
    \[
        \left|\frac{1}{2} \left(e^{2\pi i k g(N)} + e^{2\pi i k g(1)}\right)\right| \leq 1
    \]
    por lo que, tras dividir por $N$, támbien tiende a $0$.

    Por último, la segunda integral, usando que $|\partfrac{x} - \frac{1}{2}| \leq \frac{1}{2}$
    y que $|e^{2\pi i k g(x)}| = 1$, se tiene
    \[
        \left|\int_1^N \left(\partfrac{x} - \frac{1}{2}\right) 2\pi i k g'(x) \, e^{2\pi i k g(x)} \, dx\right|
        \leq \pi |k| \int_1^N  |g'(x)| \, dx
    \]
    % Aqui habria que transformar esto para ponerlo en funcion de g' para usar lim g' = 0
    Como $\lim_{x \to \infty} g'(x) = 0$, entonces para cada $\varepsilon > 0$ 
    existe $x_1 > 0$ tal que $|g'(x)| < \varepsilon$ para $x > x_1$. 

    Sea $\varepsilon > 0$ fijo y sea $N > x_1$, como $g'(x)$ es continua para $x \geq 1$, entonces
    es integrable en $[1,x_1]$, por lo que existe $M > 0$ tal que $\int_1^{x_1} |g'(x)| \, dx \leq M$.

    Por lo tanto,
    \[
        \int_1^N |g'(x)| \, dx = 
        \int_1^{x_1} |g'(x)| \, dx + \int_{x_1}^N |g'(x)| \, dx \leq 
        M + \varepsilon (N - x_1) \leq
        M + N \varepsilon
    \]
    luego,
    \[
        \limsup_{N \to \infty} \frac{1}{N} \int_1^N |g'(x)| \, dx \leq \varepsilon
    \]

    Dado que $\varepsilon > 0$ es arbitrario, y que el integrando es positivo, se concluye que
    \[
        \lim_{N \to \infty} \frac{1}{N} \int_1^N |g'(x)| \, dx = 0
    \]
    y por lo tanto,
    \[
        \lim_{N \to \infty} \frac{1}{N} \int_1^N \left(\partfrac{x} - \frac{1}{2}\right) 2\pi i k g'(x) \, e^{2\pi i k g(x)} \, dx = 0
    \]

    Todos los términos divididos por $N$ tienden a $0$ cuando $N \to \infty$, por lo tanto se cumple el 
    criterio de Weyl y la sucesión $(g(n))$ es equidistribuida módulo $1$.
\end{proof}
