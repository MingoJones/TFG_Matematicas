\begin{theorem}[Criterio de Weyl]
    \label{teo:criterio_weyl}
    Sea $(x_n)_{n\geq1}$ una sucesión de números reales, entonces son equivalentes:
    \begin{enumerate}
        \item $(x_n)$ es equidistribuida módulo $1$.
        \item Para cada $k \in \mathbb{Z} \setminus \{0\}$,
              \[
                  \frac{1}{N} \sum_{n=1}^N e^{2\pi i k x_n} \longrightarrow 0 \quad \text{cuando } N \to \infty.
              \]
    \end{enumerate}
\end{theorem}

\begin{proof}
    $1 \Longrightarrow 2$ \, Sea $k \in \mathbb{Z} \setminus \{0\}$ y sea $f(x)=e^{2\pi i k x}$, 
    $f \in C(\mathbb{T})$ y por el corolario \ref{carac:continua}, se cumple que
    \[
        \lim_{N \to \infty} \frac{1}{N} \sum_{n=1}^N e^{2\pi i k x_n} = \int_0^1 e^{2\pi i k x}\ dx = 0.
    \]
    
    $2 \Longrightarrow 1$ \, La hipótesis es el paso $2$ del lema \ref{lem:lema_na}. 
    Reproduciendo la prueba, se concluye que para cada función $f \in C(\mathbb{T})$
    \[
        \lim_{N \to \infty} \frac{1}{N} \sum_{n=1}^N f(x_n) = \int_0^1 f(x)\ dx
    \]
    y por el corolario \ref{carac:continua}, la sucesión $(x_n)$ es equidistribuida módulo $1$.
\end{proof}
